# Default options
rmdir	rmdir /s /q

# Applications
# confirm; even if exists
chin	choco install -y --force --ignore-checksums $1
chout	choco uninstall -y $1
chup	choco upgrade -y $1
chinfo	choco info $1
chls 	choco search $1

# Git & GitHub
c	git commit -m "Bulk commit" -a
p	git push

# Custom
x	exit
xb	exit /b
sqlite	sqlite3 $*

# Linux legacy
clear	cls
pwd	cd
man	help $1
rm	del $*
# eg: alias pwd=cd
alias	doskey $*
ls	dir $*
la	dir /ah
cat	type $*
mv	move $*
cp	cp $*
read	set /p $1=
passwd	net user %username% *
useradd	net user /add $1 $2
userdel	net user /del $1
usermod	net user /active:yes $1
history	doskey /history
ps	tasklist
kill	taskkill /f /im $1.exe
killall	taskkill /t /f /im $1.exe
ln	mklink $1 $2
arch	echo %processor_architecture%
grep	find "$1" $2
export	setx
uname	systeminfo
less	more
most	more
cd~	cd /d %HOME%
env	set
